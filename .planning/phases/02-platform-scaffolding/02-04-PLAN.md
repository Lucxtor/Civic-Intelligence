Create the admin dimensions API route at `src/app/api/admin/dimensions/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { db } from '@/lib/db';

// GET /api/admin/dimensions — list all dimensions
export async function GET() {
  const dims = await db.demographicDimension.findMany({
    orderBy: { createdAt: 'asc' },
  });
  return NextResponse.json(dims.map((d) => ({ ...d, options: JSON.parse(d.options) })));
}

// POST /api/admin/dimensions — create new dimension
// Body: { key: string, label: string, options: string[] }
export async function POST(req: NextRequest) {
  const { key, label, options } = await req.json();
  if (!key || !label || !options?.length) {
    return NextResponse.json({ error: 'Missing fields' }, { status: 400 });
  }
  const dim = await db.demographicDimension.create({
    data: { key, label, options: JSON.stringify(options) },
  });
  return NextResponse.json({ ...dim, options: JSON.parse(dim.options) }, { status: 201 });
}

// DELETE /api/admin/dimensions?id=...
export async function DELETE(req: NextRequest) {
  const id = req.nextUrl.searchParams.get('id');
  if (!id) return NextResponse.json({ error: 'Missing id' }, { status: 400 });
  await db.demographicDimension.delete({ where: { id } });
  return NextResponse.json({ success: true });
}
```
