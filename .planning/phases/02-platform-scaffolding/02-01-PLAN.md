## SQL Schema (for Supabase dashboard)

The following SQL must be run in the Supabase SQL editor to create the required tables. Include this as a comment block in `src/lib/supabase.ts`:

```sql
-- profiles table
create table if not exists profiles (
  wallet_address text primary key,
  demographics   jsonb not null default '{}'::jsonb,
  created_at     timestamptz default now()
);

-- demographic_dimensions table
create table if not exists demographic_dimensions (
  id         uuid primary key default gen_random_uuid(),
  key        text not null unique,
  label      text not null,
  options    jsonb not null default '[]'::jsonb,
  created_at timestamptz default now()
);

-- Seed default dimensions
insert into demographic_dimensions (key, label, options) values
  ('ageGroup',    'Age Group',    '["18–24","25–34","35–44","45–54","55+"]'),
  ('occupation',  'Occupation',   '["Tech Worker","Student","Educator","Public Servant","Business Owner","Other"]'),
  ('district',    'District',     '["Centro","Trindade","Lagoa","Campeche","Continente","Other"]')
on conflict (key) do nothing;
```
