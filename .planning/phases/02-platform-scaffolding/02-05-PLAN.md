<action>
Create `src/app/api/vote/route.ts` — a POST handler that:
1. Receives `{ proposalId: string, walletAddress: string, responses: Record<string, number> }`.
2. Uses `createServerClient()` (service role) to look up `demographics` from `profiles` where `wallet_address = walletAddress`.
3. Writes `{ proposal_id: proposalId, demographics, responses, created_at }` to a `votes` table (anonymized — no wallet address stored).
4. Returns `{ success: true }` or an error response.

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { createServerClient } from '@/lib/supabase-server';

export async function POST(req: NextRequest) {
  const { proposalId, walletAddress, responses } = await req.json();

  if (!proposalId || !walletAddress || !responses) {
    return NextResponse.json({ error: 'Missing fields' }, { status: 400 });
  }

  const supabase = createServerClient();

  // Fetch demographics by wallet (server-side only — never returned to client)
  const { data: profile, error: profileErr } = await supabase
    .from('profiles')
    .select('demographics')
    .eq('wallet_address', walletAddress)
    .single();

  if (profileErr || !profile) {
    return NextResponse.json({ error: 'Profile not found — complete onboarding first.' }, { status: 404 });
  }

  // Write anonymous vote (no wallet address)
  const { error: voteErr } = await supabase
    .from('votes')
    .insert({ proposal_id: proposalId, demographics: profile.demographics, responses });

  if (voteErr) {
    return NextResponse.json({ error: voteErr.message }, { status: 500 });
  }

  return NextResponse.json({ success: true });
}
```

Also add the `votes` table SQL as a comment:
```sql
create table if not exists votes (
  id          uuid primary key default gen_random_uuid(),
  proposal_id text not null,
  demographics jsonb not null,
  responses    jsonb not null,
  created_at  timestamptz default now()
);
```
