Create `src/app/api/vote/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { db } from '@/lib/db';

// POST /api/vote
// Body: { proposalId: string, walletAddress: string, responses: Record<string, number> }
export async function POST(req: NextRequest) {
  const { proposalId, walletAddress, responses } = await req.json();

  if (!proposalId || !walletAddress || !responses) {
    return NextResponse.json({ error: 'Missing required fields' }, { status: 400 });
  }

  // Fetch demographics by walletAddress — this is the only time the address is used
  const user = await db.user.findUnique({ where: { walletAddress } });

  if (!user) {
    return NextResponse.json(
      { error: 'Profile not found. Please complete onboarding first.' },
      { status: 404 }
    );
  }

  // Write anonymized vote — NO walletAddress stored here
  await db.anonymizedVote.create({
    data: {
      proposalId,
      demographics: user.demographics, // already a JSON string
      responses: JSON.stringify(responses),
    },
  });

  return NextResponse.json({ success: true });
}
```
